        <h4>Sketchbook</h4>
        <canvas id="sketchCanvas" width="600" height="400" style="border: 1px solid black;"></canvas>
        <br/>
        <button id="eraserButton">Eraser</button>
        <button id="penButton">Pen</button>
        <button id="clearButton">Clear</button>
    </div>

    <script>
        const canvas = document.getElementById('sketchCanvas');
        const ctx = canvas.getContext('2d');
        let isDrawing = false;
        let isErasing = false;

        canvas.addEventListener('mousedown', (e) => {
            isDrawing = true;
            draw(e);
        });

        canvas.addEventListener('mouseup', () => {
            isDrawing = false;
            ctx.beginPath();
        });

        canvas.addEventListener('mousemove', (e) => {
            if (!isDrawing) return;
            draw(e);
        });

        function draw(e) {
            const x = e.offsetX;
            const y = e.offsetY;
            if (isErasing) {
                ctx.globalCompositeOperation = 'destination-out';
                ctx.fillStyle = 'rgba(0,0,0,1)';
                ctx.beginPath();
                ctx.arc(x, y, 10, 0, 2 * Math.PI);
                ctx.fill();
            } else {
                ctx.globalCompositeOperation = 'source-over';
                ctx.lineWidth = 2;
                ctx.lineCap = 'round';
                ctx.strokeStyle = 'black';
                ctx.lineTo(x, y);
                ctx.stroke();
                ctx.beginPath();
                ctx.moveTo(x, y);
            }
        }

        document.getElementById('eraserButton').addEventListener('click', () => {
            isErasing = true;
        });

        document.getElementById('penButton').addEventListener('click', () => {
            isErasing = false;
        });

        document.getElementById('clearButton').addEventListener('click', () => {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
        });
    </script>

        <h2>DNO Game</h2>
        <p>This is a simple game where you have to guess if a number is divisible by 3, 5, or neither.</p>
        <input type="number" id="numberInput" placeholder="Enter a number">
        <button id="checkButton">Check</button>
        <p id="result"></p>

        <script>
            document.getElementById('checkButton').addEventListener('click', function() {
                const number = document.getElementById('numberInput').value;
                let result = '';

                if (number % 3 === 0) {
                    result += 'Divisible by 3. ';
                }
                if (number % 5 === 0) {
                    result += 'Divisible by 5. ';
                }
                if (result === '') {
                    result = 'Not divisible by 3 or 5.';
                }

                document.getElementById('result').textContent = result;
            });
        </script>
